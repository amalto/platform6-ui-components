<Story
  name="Form"
  parameters={{ docs: { disable: true } }}
  decorators={[withKnobs]}
>
  {() => {
    const values = decorate([
      (args) =>
        Array.from(args[0].target.elements)
          .filter((i) => !!i.name)
          .map((i) => ({ name: i.name, value: i.value })),
    ]);
    const kPattern = kText("Pattern");
    const kReadonly = boolean("Readonly", false);
    const kRequired = boolean("Required", true);
    const kDisabled = boolean("Disabled", false);
    const kWaiting = boolean("Waiting", false);
    const formContent = `
      <p6-field>
        <p6-label slot="label">A label</p6-label>
        <p6-input
        name="test"
        disabled=${kDisabled}
        pattern=${kPattern}
        readonly=${kReadonly}
        required=${kRequired}
        waiting=${kWaiting}></p6-input>
      </p6-field>
      <div class="field">
        <div class="control">
          <p6-button class="button is-link">Submit</p6-button>
        </div>
      </div>
    `;
    const form = document.createElement("form");
    form.noValidate = true;
    form.addEventListener("submit", (e) => {
      e.preventDefault();
      values.action("values")(e);
      action("event")(e);
    });
    form.innerHTML = formContent;
    return form;
  }}
</Story>
