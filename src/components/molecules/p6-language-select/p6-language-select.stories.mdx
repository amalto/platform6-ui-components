# Language Select

`p6-language-select` provides a select

<Story
  name="Default"
  parameters={{ docs: { disable: true } }}
  decorators={[withKnobs]}
>
  {() => {
    const kLocales = select("Language", ["en", "fr", "ru"], "en");
    const kDisabled = boolean("Disabled", false);
    const kValue = text("Value", "en");
    const kExcludes = text("Exclude", "fr");
    const kModes = select(
      "Mode",
      ["default", "primary", "info", "success", "warning", "danger"],
      "default"
    );
    return `
      <p6-language-select
        name="test"
        lang="${kLocales}"
        disabled="${kDisabled}"
        value="${kValue}"
        excludes="${kExcludes}"
        mode="${kModes}"
        />
    `;
  }}
</Story>
