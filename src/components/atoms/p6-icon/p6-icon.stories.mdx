# Icon

`p6-icon` provides icon from [Font Awesome](https://fontawesome.com/)

<Story
  name="Default"
  parameters={{ docs: { disable: true } }}
  decorators={[withKnobs]}
>
  {() => {
    library.add(
      fab.faAccessibleIcon,
      fab.faAutoprefixer,
      fab.faFontAwesome,
      fab.faGit,
      fab.faHtml5,
      fas.faCat,
      fas.faInfoCircle,
      fas.faCheckSquare,
      fas.faHome,
      fas.faTimes
    );
    const names = [
      "accessible-icon",
      "autoprefixer",
      "font-awesome",
      "git",
      "html5",
      "cat",
      "info-circle",
      "check-square",
      "home",
      "times",
    ];
    const prefix = ["fab", "fas"];
    const kPrefix = select("Style", ["fab", "fas"], "fab");
    const kName = select("Name", names, "font-awesome");
    const kTransform = text("Transformation");
    return `<p6-icon name=${kName} icon-prefix=${kPrefix} transform=${kTransform}></p6-icon>`;
  }}
</Story>
