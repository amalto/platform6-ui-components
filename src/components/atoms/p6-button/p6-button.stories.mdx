# Button

<Story
  name="Default"
  parameters={{ docs: { disable: true } }}
  decorators={[withKnobs]}
>
  {(el) => {
    const modes = [
      "danger",
      "warning",
      "default",
      "info",
      "success",
      "primary",
    ];
    const sizes = ["small", "default", "normal", "medium", "large"];
    const kModes = select("Modes", modes, "default");
    const kSize = select("Size", sizes, "default");
    const kOutlined = boolean("Outlined", false);
    const kWaiting = boolean("Waiting", false);
    const kDisabled = boolean("Disabled", false);
    const component = document.createElement("p6-button");
    component.mode = kModes;
    component.waiting = kWaiting;
    component.disabled = kDisabled;
    component.size = kSize;
    component.outlined = kOutlined;
    component.addEventListener("click", action("Click"));
    component.innerHTML = `Click!`;
    return component;
  }}
</Story>
