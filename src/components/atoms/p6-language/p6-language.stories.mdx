# Select

`p6-language` provides a select

<Story
  name="Default"
  parameters={{ docs: { disable: true } }}
  decorators={[withKnobs]}
>
  {() => {
    const values = decorate([
      (args) => {
        var values = [];
        new FormData(args[0].target).forEach(function (value, name) {
          values.push({ name, value });
        });
        return values;
      },
    ]);
    const kSizes = select(
      "Size",
      ["default", "small", "normal", "medium", "large"],
      "default"
    );
    const kRequired = boolean("Required", true);
    const kDisabled = boolean("Disabled", false);
    const kReadOnly = boolean("Read only", true);
    const kValue = text("Value", "fr");
    const formContent = `
      <p6-language
        name="test"
        disabled="${kDisabled}"
        required="${kRequired}"
        readOnly="${kReadOnly}"
        size="${kSizes}"
        value="${kValue}"
        />
      <div class="field">
        <div class="control">
          <p6-button class="button is-link">Submit</p6-button>
          <p6-button class="button is-link" type="reset">Reset</p6-button>
        </div>
      </div>
    `;
    const form = document.createElement("form");
    //form.noValidate = true;
    form.addEventListener("submit", (e) => {
      e.preventDefault();
      values.action("values")(e);
      action("event")(e);
    });
    form.innerHTML = formContent;
    return form;
  }}
</Story>
